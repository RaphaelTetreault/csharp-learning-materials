using System;

class Program
{
    static void Main()
    {
        // Declare a variable to store score.
        // Assign it the value of 10,000.
        int score = 10000;
        Console.WriteLine($"The score value is '{score}'");
        
        // Multiply score by 5
        score *= 5;
        Console.WriteLine($"The score value is '{score}'");
        
        // Multiply score by 2
        score *= 2;
        Console.WriteLine($"The score value is '{score}'");
        
        // Multiply score by 10,000
        score *= 10000;
        Console.WriteLine($"The score value is '{score}'");
    }
}
